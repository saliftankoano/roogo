\section{Positionnement (simple \& solide)}
\textbf{Niche : Toyota \& Lexus} avec un focus sur des \textbf{consommables \`a forte rotation} et des \textbf{pi\`eces \`a pannes r\'ecurrentes} adapt\'ees aux conditions (routes, chaleur, poussi\`ere) du Burkina Faso.

\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Pourquoi Toyota \& Lexus} : parc important, forte demande, clients r\'ecurrents (garages, flottes) + recherche de qualit\'e (Lexus).
  \item \textbf{Avantage gagnant} : disponibilit\'e r\'eelle des pi\`eces dont les gens ont besoin \emph{tout de suite}, + aide au bon choix (compatibilit\'e).
  \item \textbf{Clients prioritaires} : garages/m\'ecaniciens (B2B), flottes (taxis, livraison, ONG), et particuliers.
\end{itemize}

\section{Mod\`eles cibles (d\'emarrage)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Toyota Hilux} (pick-up; suspension, refroidissement, freins, transmission)
  \item \textbf{Toyota Land Cruiser} (s\'eries 70/80/100; suspension lourde, refroidissement, transmission)
  \item \textbf{Toyota Hiace} (utilitaire; freins, refroidissement, direction, \'electricit\'e)
  \item \textbf{Toyota Corolla / Camry} (gros volumes; allumage, suspension, freins, capteurs)
  \item \textbf{Lexus RX / GX / LX} (forte demande; pi\`eces plus ch\`eres \& souvent rares)
\end{itemize}

\section{Plan d'approvisionnement (Alibaba, bien fait)}
\subsection{R\`egle \#1 : jamais de gros volume \`a l'aveugle}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{\'Echantillons d'abord} : commander 2--5 unit\'es par r\'ef\'erence critique.
  \item \textbf{Test avec des garages partenaires} : montage sur vrais v\'ehicules; retour en 2--4 semaines.
  \item \textbf{Puis achat en volume} : constituer une liste courte de fournisseurs fiables.
\end{itemize}

\subsection{Checklist de s\'election des fournisseurs}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Sp\'ecialisation} : Toyota/Lexus aftermarket (pas uniquement \og trading \fg{})
  \item \textbf{R\'egularit\'e} : m\^eme usine, r\'ef\'erences stables, emballage coh\'erent
  \item \textbf{Documents} : factures, contr\^ole qualit\'e, conditions de garantie claires
  \item \textbf{MOQ flexible} : cartons mixtes (plusieurs r\'ef\'erences) au d\'emarrage
\end{itemize}

\subsection{Gammes qualit\'e (prot\'eger la r\'eputation)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Gamme A (recommand\'ee)} : moins de retours, prix plus \'elev\'e, id\'eal garages/flottes
  \item \textbf{Gamme B (valeur)} : moins ch\`ere mais acceptable; transparence obligatoire
  \item \'Eviter l'ultra-bon-march\'e pour les pi\`eces de s\'ecurit\'e (freins, direction).
\end{itemize}

\section{Strat\'egie de stock (peu de risque cash, forte disponibilit\'e)}
\subsection{La m\'ethode 80/20}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item D\'emarrer avec une liste courte qui couvre \textbf{la plupart des besoins hebdomadaires}.
  \item Noter \textbf{chaque demande non servie} : c'est votre feuille de route d'extension.
\end{itemize}

\subsection{Stock en deux paniers}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{\`A stocker en permanence} (forte rotation) : filtres, plaquettes, biellettes, silentblocs, roulements courants, thermostats.
  \item \textbf{Sur commande} (lent/cher) : injecteurs, calculateurs, capteurs rares, pi\`eces de carrosserie sp\'ecifiques.
\end{itemize}

\subsection{Contr\^ole de compatibilit\'e (\'evite les erreurs)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item Collecter : mod\`ele + ann\'ee + code moteur (si possible) + photo de l'ancienne pi\`ece + VIN (si dispo).
  \item Ranger par \textbf{familles de plateformes} (ex. Hilux Vigo, Corolla E120/E140, etc.).
\end{itemize}

\section{Go-to-market (comment gagner localement)}
\subsection{Distribution \og garage d'abord \fg{}}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item Nouer des relations avec \textbf{10--30 garages}. Prix pro + livraison rapide.
  \item Catalogue WhatsApp : photos, prix, et statut \og en stock \fg{} mis \`a jour chaque semaine.
  \item Proposer \textbf{livraison le jour-m\^eme / J+1} en ville (marge premium).
\end{itemize}

\subsection{Packs qui se vendent (upsell simple)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Pack entretien} : filtres air + huile + carburant (avec recommandation d'huile).
  \item \textbf{Pack refroidissement} : bouchon + thermostat + durites (+ liquide si vendu).
  \item \textbf{Pack train avant} : rotules + biellettes + stabilisateurs (rappel parall\'elisme).
\end{itemize}

\section{Prix \& garantie (la confiance = clients r\'ecurrents)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Deux niveaux de prix} : \og Valeur \fg{} vs \og Recommand\'e \fg{}.
  \item R\`egles de garantie simples (ex. \textbf{montage par m\'ecanicien}, \textbf{pas de casse}, ticket obligatoire).
  \item Suivre les retours par fournisseur pour prot\'eger la tr\'esorerie et la r\'eputation.
\end{itemize}

\subsection{Plan de croissance}
\begin{enumerate}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Mois 1--2} : vendre le stock initial; r\'einvestir 100\% du b\'en\'efice brut en r\'eappro.
  \item \textbf{Mois 3+} : une fois le cash-flow positif, regrouper les fonds pour une premi\`ere commande \'echantillon Alibaba (2--5 unit\'es/SKU) afin de tester la qualit\'e.
  \item \textbf{Mois 6+} : si les \'echantillons passent, faire une premi\`ere importation en volume (\texttildelow3--5M XOF) financ\'ee par les b\'en\'efices accumul\'es.
\end{enumerate}

\subsection{Facteurs cl\'es de succ\`es \`a ce budget}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Catalogue WhatsApp} mis \`a jour quotidiennement (photos + prix + \og en stock \fg{}).
  \item \textbf{Relations avec 3--5 garages} qui ach\`etent chaque semaine.
  \item \textbf{Suivre chaque demande non satisfaite} : c'est votre liste de r\'eappro.
  \item \textbf{Z\'ero stock mort} : ne r\'eapprovisionner que ce qui s'est vendu les 2 derni\`eres semaines.
\end{itemize}
