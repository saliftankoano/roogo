\section{Positioning (Simple \& Strong)}
\textbf{Niche: Toyota-only} parts focused on \textbf{high-demand consumables and chronic-failure components} for Burkina Faso road/heat/dust conditions.

\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Why Toyota}: dominant market share, large installed base, easy customer targeting, repeat buyers (mechanics, fleets).
  \item \textbf{Winning edge}: dependable availability of the parts people need \emph{this week}, plus correct fitment guidance.
  \item \textbf{Primary customers}: garages/mechanics (B2B), fleet owners (taxis, deliveries, NGOs), and walk-in owners.
\end{itemize}

\section{Target Models (Start Here)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Hilux} (workhorse pickups; suspension, cooling, brakes, drivetrain)
  \item \textbf{Land Cruiser} (70/80/100 series; heavy-duty suspension, cooling, drivetrain)
  \item \textbf{Hiace} (commercial vans; brakes, cooling, steering, electrical)
  \item \textbf{Corolla / Camry} (high volume; ignition, suspension, brakes, sensors)
\end{itemize}

\subsection{Next Wave (High Margin / High Pain)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Timing belt kits} (belt + tensioner + idler; optionally water pump)
  \item \textbf{Engine mounts} (rubber degrades in heat; reduces vibration complaints)
  \item \textbf{Fuel pumps / injectors} (dirty fuel; offer diagnostics + warranty terms)
  \item \textbf{CV joints / driveshaft components} (Hilux/Land Cruiser/Hiace)
  \item \textbf{Cooling sensors} (coolant temp sensor, fan switch), \textbf{fan clutch} / electric fan assemblies
  \item \textbf{Ignition coils} + spark plugs (Corolla/Camry high volume)
  \item \textbf{ABS / wheel speed sensors} (dust and cable damage)
\end{itemize}

\subsection{Body/Convenience (Fast Cash, Low Complexity)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item Mirrors, headlights/tail lights, door handles, window switches, wiper motors
  \item These sell well when you have \textbf{correct fitment} and \textbf{stock on hand}

\section{Sourcing Plan (Alibaba Done Right)}
\subsection{Rule \#1: No blind bulk orders}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Sample first}: order 2--5 units of each critical SKU.
  \item \textbf{Test with partner garages}: install on real cars; collect feedback in 2--4 weeks.
  \item \textbf{Only then buy bulk}: build a short list of trusted suppliers.
\end{itemize}

\subsection{Supplier selection checklist}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Specialization}: supplier focuses on Toyota aftermarket parts (not general trading only)
  \item \textbf{Consistency}: same factory, stable part numbers, repeatable packaging
  \item \textbf{Documentation}: invoices, QC reports, clear warranty terms
  \item \textbf{MOQ flexibility}: accept mixed-SKU cartons for early-stage inventory
\end{itemize}

\subsection{Quality tiers (avoid reputation damage)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Tier A (recommended)}: fewer returns, higher price, best for mechanics/fleets
  \item \textbf{Tier B (value)}: lower price, still acceptable; be transparent
  \item Avoid ultra-cheap parts for safety-critical items (brakes, steering).
\end{itemize}

\section{Inventory Strategy (Low Cash Risk, High Availability)}
\subsection{The 80/20 method}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item Start with a focused list that covers \textbf{most weekly needs}.
  \item Track \textbf{every request you cannot fill}; that list becomes your expansion roadmap.
\end{itemize}

\subsection{Two-bucket stocking}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Always-stock} (fast movers): filters, pads, links, bushings, common bearings, thermostats.
  \item \textbf{Order-on-request} (slow movers/expensive): injectors, ECUs, rare sensors, specific body trims.
\end{itemize}

\subsection{Fitment control (prevents wrong-part losses)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item Collect: model + year + engine code (when possible) + photo of old part + VIN (when available).
  \item Label shelves by \textbf{platform codes} (e.g., Hilux Vigo era, Corolla E120/E140, etc.).
\end{itemize}

\section{Go-to-Market (How You Win Locally)}
\subsection{Mechanic-first distribution}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item Build relationships with \textbf{10--30 garages}. Offer trade pricing and quick delivery.
  \item Create a WhatsApp catalog: photos, prices, and ``in stock'' status updated weekly.
  \item Offer \textbf{same-day / next-day delivery} inside the city for premium pricing.
\end{itemize}

\subsection{Bundles that sell (simple upsells)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item \textbf{Service kits}: air + oil + fuel filters; add oil grade recommendations.
  \item \textbf{Cooling kit}: radiator cap + thermostat + hoses + coolant (if sold).
  \item \textbf{Front-end kit}: ball joints + tie rods + stabilizer links (plus alignment reminder).
\end{itemize}

\section{Pricing \& Warranty (Trust = Repeat Customers)}
\begin{itemize}[leftmargin=*,itemsep=0.25em]
  \item Use \textbf{two-tier pricing}: Value vs Recommended.
  \item Publish clear warranty rules (e.g., \textbf{installation by mechanic}, \textbf{no physical damage}, keep receipt).
  \item Track returns by supplier to protect your cash and reputation.
\end{itemize}

\section{30-Day Launch Checklist}
\begin{enumerate}[leftmargin=*,itemsep=0.25em]
  \item Choose 2--4 target Toyota models and lock the initial SKU list (Top 12 + service kits).
  \item Identify 3--5 suppliers per category; order samples; test with partner garages.
  \item Create a simple WhatsApp catalog and mechanic price sheet.
  \item Set delivery plan (motorbike courier) and a returns/warranty policy.
  \item Open with ``always-stock'' items and add new SKUs weekly based on requests.
\end{enumerate}
