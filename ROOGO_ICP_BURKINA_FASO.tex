\section{Introduction}
Ce document définit les profils clients idéaux (ICP) pour Roogo au Burkina Faso. Ces segments cibles guideront notre stratégie marketing, notre approche d'influence marketing et notre création de contenu sur les réseaux sociaux.

\section{Profil Client Idéal (ICP) - Burkina Faso}

\subsection{ICP 1 : Chercheurs de Propriété (Revenus Early Bird)}

\subsubsection{Démographie}
\begin{itemize}
    \item \textbf{Âge :} 25-40 ans
    \item \textbf{Localisation :} Centres urbains (Ouagadougou, Bobo-Dioulasso, Koudougou)
    \item \textbf{Revenu :} 150 000 - 500 000 XOF/mois (classe moyenne à moyenne supérieure)
    \item \textbf{Profession :} Jeunes professionnels, entrepreneurs, cadres moyens, expatriés ONGs/organisations internationales
    \item \textbf{Éducation :} Universitaire ou professionnels qualifiés
    \item \textbf{Situation familiale :} Jeunes couples, petites familles, professionnels célibataires en relocalisation
\end{itemize}

\subsubsection{Psychographie}
\begin{itemize}
    \item \textbf{Valeurs :} Efficacité temporelle, transparence, solutions modernes, éviter les tracas
    \item \textbf{Points de Douleur :}
    \begin{itemize}
        \item Fatigués des propriétaires et agents peu fiables
        \item Argent perdu sur fausses annonces ou absences
        \item Week-ends gaspillés à visiter des biens inadaptés
        \item Peur de manquer les bonnes opportunités
    \end{itemize}
    \item \textbf{Comportement :}
    \begin{itemize}
        \item Mobile-first, actifs sur réseaux sociaux
        \item Valorisent la commodité et prêts à payer pour
        \item Décisions rapides face à options de qualité
        \item Recherchent en ligne avant décision
    \end{itemize}
\end{itemize}

\subsubsection{Où les Trouver}
\textbf{Lieux Physiques :}
\begin{itemize}
    \item Cafés modernes/espaces de coworking (Zone du Bois, Ouaga 2000)
    \item Écoles internationales
    \item Centres commerciaux (Marina Market, SIAO)
    \item Centres de fitness et gymnases
    \item Campus universitaires (diplômés récents)
\end{itemize}

\textbf{Canaux Digitaux :}
\begin{itemize}
    \item Groupes Facebook : "Expats Ouagadougou", "Location Ouaga", "Appartements Burkina"
    \item Instagram (scrollers actifs, viewers stories)
    \item Groupes WhatsApp (réseaux professionnels, amis)
    \item LinkedIn (relocalisations professionnelles)
    \item Google : "appartement ouagadougou", "location maison burkina"
\end{itemize}

\subsection{ICP 2 : Propriétaires (Source Principale de Revenus)}

\subsubsection{Démographie}
\begin{itemize}
    \item \textbf{Âge :} 35-60 ans
    \item \textbf{Localisation :} Propriétés à Ouagadougou, Bobo-Dioulasso, quartiers émergents
    \item \textbf{Revenu :} 500 000+ XOF/mois, souvent multiples sources
    \item \textbf{Profession :} Chefs d'entreprise, cadres supérieurs, investisseurs, diaspora
    \item \textbf{Portfolio :} 1-5 propriétés (petits à moyens propriétaires)
    \item \textbf{Technologie :} Modéré à élevé (à l'aise avec paiements mobiles)
\end{itemize}

\subsubsection{Psychographie}
\begin{itemize}
    \item \textbf{Valeurs :} ROI, efficacité, service professionnel, réduction vacance
    \item \textbf{Points de Douleur :}
    \begin{itemize}
        \item Propriétés vacantes réduisant profits
        \item Agents traditionnels peu fiables prenant grandes commissions
        \item Temps perdu sur prospects non qualifiés
        \item Difficulté à gérer plusieurs propriétés
        \item Préoccupations dommages par mauvais locataires
    \end{itemize}
    \item \textbf{Comportement :}
    \begin{itemize}
        \item Orientés résultats, prêts à investir pour meilleurs résultats
        \item Valorisent hautement leur temps
        \item Veulent locataires qualifiés et sérieux
        \item Apprécient présentation professionnelle de leurs actifs
    \end{itemize}
\end{itemize}

\subsubsection{Où les Trouver}
\textbf{Lieux Physiques :}
\begin{itemize}
    \item Événements networking business
    \item Rassemblements Chambre de Commerce
    \item Événements développement immobilier
    \item Restaurants/hôtels haut de gamme (déjeuners d'affaires)
    \item Banques (départements investissement/prêt)
\end{itemize}

\textbf{Canaux Digitaux :}
\begin{itemize}
    \item Groupes Facebook : "Investissement immobilier Burkina", "Entrepreneurs BF"
    \item LinkedIn (professionnels business)
    \item Réseaux WhatsApp business
    \item Forums et communautés immobilier
\end{itemize}

\subsection{ICP 3 : Early Adopters (Critique pour Phase Lancement)}

\subsubsection{Profil}
\begin{itemize}
    \item \textbf{Âge :} 22-35 ans
    \item \textbf{Profil :} Passionnés tech, écosystème startup, early adopters
    \item \textbf{Valeurs :} Être premier, innovation, soutien startups locales
    \item \textbf{Comportement :}
    \begin{itemize}
        \item Actifs réseaux sociaux, créent contenu
        \item Influencent amis et famille
        \item Prêts à tester nouvelles plateformes
        \item Fournissent feedback et engagent avec marques
    \end{itemize}
\end{itemize}

\section{Stratégie Marketing d'Influence}

\subsection{Tier 1 : Micro-Influenceurs (5K - 50K abonnés)}
\textbf{Profil :}
\begin{itemize}
    \item Jeunes professionnels partageant lifestyle
    \item Créateurs contenu immobilier/investissement
    \item Influenceurs expats documentant vie au Burkina
    \item Comédiens/divertissement locaux (portée virale)
\end{itemize}

\textbf{Types de Contenu :}
\begin{itemize}
    \item Parcours authentique recherche appartement avec Roogo
    \item Storytelling "Ma recherche était facile"
    \item Sketches comiques recherche traditionnelle vs Roogo
\end{itemize}

\subsection{Tier 2 : Influenceurs Business/Professionnels}
\textbf{Profil :}
\begin{itemize}
    \item Consultants business avec présence LinkedIn
    \item Conseillers investissement immobilier
    \item Éducateurs littératie financière
\end{itemize}

\textbf{Types de Contenu :}
\begin{itemize}
    \item Conseils investissement avec Roogo comme étude de cas
    \item Contenu efficacité gestion immobilière
    \item Calculs ROI pour propriétaires
\end{itemize}

\subsection{Tier 3 : Leaders Communautaires}
\textbf{Profil :}
\begin{itemize}
    \item Leaders associations étudiantes universitaires
    \item Présidents associations professionnelles
    \item Chefs associations de quartier
\end{itemize}

\section{Piliers de Contenu Réseaux Sociaux}

\subsection{Pilier 1 : Preuve Sociale \& FOMO (40\%)}
\begin{itemize}
    \item Propriétés verrouillées en heures
    \item Nombre candidatures reçues
    \item Success stories et témoignages
    \item Comptes à rebours propriétés populaires
\end{itemize}

\subsection{Pilier 2 : Éducation \& Transparence (30\%)}
\begin{itemize}
    \item Fonctionnement plateforme
    \item Détails tarification
    \item Guides quartiers
    \item Droits location et conseils
    \item Coulisses photographie professionnelle
\end{itemize}

\subsection{Pilier 3 : Lifestyle \& Aspiration (20\%)}
\begin{itemize}
    \item Vitrines belles propriétés
    \item Vie moderne au Burkina Faso
    \item Contenu "Maison de rêve"
    \item Inspiration design intérieur
\end{itemize}

\subsection{Pilier 4 : Communauté \& Engagement (10\%)}
\begin{itemize}
    \item Contenu généré utilisateurs
    \item Sondages et questions
    \item Réponses actualités marché immobilier
    \item Célébration jalons (propriétés listées/louées)
\end{itemize}

\section{Stratégie par Canal}

\subsection{Facebook}
\begin{itemize}
    \item \textbf{Audience :} Propriétaires (35-60), chercheurs généraux
    \item \textbf{Contenu :} Posts détaillés, études de cas, groupes, publicités ciblées
    \item \textbf{Objectif :} Génération leads, éducation, construction communauté
\end{itemize}

\subsection{Instagram}
\begin{itemize}
    \item \textbf{Audience :} Jeunes professionnels (25-35), lifestyle
    \item \textbf{Contenu :} Reels (tours propriétés, conseils rapides), Stories (comptes à rebours, sondages), carrousels (avant/après)
    \item \textbf{Objectif :} Notoriété marque, création FOMO, engagement
\end{itemize}

\subsection{WhatsApp}
\begin{itemize}
    \item \textbf{Audience :} Les deux ICP - canal communication directe
    \item \textbf{Contenu :} Mises à jour personnalisées, nouvelles annonces, alertes early bird exclusives
    \item \textbf{Objectif :} Conversion, rétention, communication personnalisée
\end{itemize}

\subsection{TikTok}
\begin{itemize}
    \item \textbf{Audience :} 20-30 ans, portée virale
    \item \textbf{Contenu :} Approches comiques recherche appartement, tours rapides, sons tendance
    \item \textbf{Objectif :} Notoriété marque, viralité, engagement jeunesse
\end{itemize}

\subsection{LinkedIn}
\begin{itemize}
    \item \textbf{Audience :} Propriétaires, professionnels business, expats
    \item \textbf{Contenu :} Insights marché, conseils investissement, études de cas professionnelles
    \item \textbf{Objectif :} Partenariats B2B, crédibilité, acquisition propriétaires
\end{itemize}
